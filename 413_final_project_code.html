def addProBowl(row):
    player_name = row['name']

def create_norm_data(data, cols_to_norm, cols_to_drop):
    n_data = data.copy()
    n_data['pro_bowl'] = n_data.apply(addProBowl, axis=1)
    n_data = n_data.drop(cols_to_drop, axis = 1)
    n_data[cols_to_norm] = n_data[cols_to_norm].apply(lambda x: (x - x.min()) / (x.max() - x.min()))
    return n_data
